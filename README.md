# Timewith Accordion Component

## Install

```bash
$ yarn add @time-with/accordion
```

## Usage

    import Accordion from '@time-with/accordion'

    const items = [
      {
        id: 'id1',
        title: 'title 1',
        html: <p>html 1</p>
      },
      {
        id: 'id2',
        title: 'title 2',
        html: <p>html 2</p>
      },
      {
        id: 'id3',
        title: 'title 3',
        html: <p>html 3</p>
      },
    ]

    <Accordion items={items}/>
